```js
// server/lib/memoryDb.js
import Database from 'better-sqlite3'
import * as sqliteVec from 'sqlite-vec'
import { randomUUID } from 'crypto'

const MEMORY_DB_PATH = process.env.MEMORY_DB_PATH || `${process.env.HOME}/.claude/memory.db`
const memDb = new Database(MEMORY_DB_PATH)
memDb.pragma('journal_mode = WAL')
sqliteVec.load(memDb)

// ---- Lesen ----

export function listMemories({ q, domain, area, wichtigkeit, schlagwort, page = 1, limit = 50 } = {}) {
  const offset = (page - 1) * limit
  const conditions = []
  const params = []

  if (q) {
    conditions.push("m.text LIKE ?")
    params.push(`%${q}%`)
  }
  if (domain) {
    conditions.push("m.domain = ?")
    params.push(domain)
  }
  if (area) {
    conditions.push("m.area LIKE ?")
    params.push(`%${area}%`)
  }
  if (wichtigkeit) {
    conditions.push("m.wichtigkeit = ?")
    params.push(wichtigkeit)
  }
  if (schlagwort) {
    conditions.push("m.schlagwoerter LIKE ?")
    params.push(`%${schlagwort}%`)
  }

  const where = conditions.length ? `WHERE ${conditions.join(' AND ')}` : ''

  const total = memDb.prepare(`SELECT COUNT(*) AS n FROM memories m ${where}`).get(...params).n
  const memories = memDb.prepare(
    `SELECT id, text, domain, area, wichtigkeit, schlagwoerter, quelle_typ, quelle, created_at, updated_at
     FROM memories m ${where}
     ORDER BY m.updated_at DESC
     LIMIT ? OFFSET ?`
  ).all(...params, limit, offset)

  return { memories, total, page, pages: Math.ceil(total / limit) }
}

export function getMemory(id) {
  return memDb.prepare(
    `SELECT id, text, domain, area, area_sek, wichtigkeit, schlagwoerter, quelle_typ, quelle, created_at, updated_at
     FROM memories WHERE id = ?`
  ).get(id) || null
}

// ---- Schreiben ----

export function insertMemory(data, embeddingBytes) {
  const id = randomUUID()
  const now = new Date().toISOString().replace('T', ' ').slice(0, 19)

  const insertFn = memDb.transaction(() => {
    memDb.prepare(`
      INSERT INTO memories (id, text, domain, area, wichtigkeit, schlagwoerter, quelle_typ, quelle, created_at, updated_at)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
    `).run(
      id,
      data.text,
      data.domain,
      data.area,
      data.wichtigkeit || 'Wichtig',
      data.schlagwoerter || null,
      'bestätigt',
      'devd-memory-ui',
      now,
      now
    )
    if (embeddingBytes) {
      memDb.prepare(
        'INSERT INTO memory_vec(memory_id, embedding) VALUES (?, ?)'
      ).run(id, embeddingBytes)
    }
  })

  insertFn()
  return getMemory(id)
}

export function updateMemory(id, data, embeddingBytes) {
  const now = new Date().toISOString().replace('T', ' ').slice(0, 19)
  const existing = getMemory(id)
  if (!existing) return null

  const updateFn = memDb.transaction(() => {
    memDb.prepare(`
      UPDATE memories SET
        text = ?,
        domain = ?,
        area = ?,
        wichtigkeit = ?,
        schlagwoerter = ?,
        updated_at = ?
      WHERE id = ?
    `).run(
      data.text ?? existing.text,
      data.domain ?? existing.domain,
      data.area ?? existing.area,
      data.wichtigkeit ?? existing.wichtigkeit,
      data.schlagwoerter !== undefined ? data.schlagwoerter : existing.schlagwoerter,
      now,
      id
    )
    if (embeddingBytes) {
      memDb.prepare('DELETE FROM memory_vec WHERE memory_id = ?').run(id)
      memDb.prepare('INSERT INTO memory_vec(memory_id, embedding) VALUES (?, ?)').run(id, embeddingBytes)
    }
  })

  updateFn()
  return getMemory(id)
}

export function deleteMemory(id) {
  const deleteFn = memDb.transaction(() => {
    memDb.prepare('DELETE FROM memory_vec WHERE memory_id = ?').run(id)
    memDb.prepare('DELETE FROM memories WHERE id = ?').run(id)
  })
  deleteFn()
}
```

```js
async function getEmbedding(text) {
  try {
    const res = await fetch(`${OLLAMA_URL}/api/embed`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ model: OLLAMA_EMBED_MODEL, input: text }),
    })
    if (!res.ok) throw new Error(`Ollama HTTP ${res.status}`)
    const data = await res.json()
    const floats = new Float32Array(data.embeddings[0])
    const { serialize } = await import('sqlite-vec')
    return serialize(floats)
  } catch (e) {
    console.warn('[memory] Ollama embed fehlgeschlagen:', e.message)
    return null
  }
}
```

```js
// ---- Memory Routes ----

// GET /api/memories — Liste mit Filtern und Pagination
app.get('/api/memories', (req, res) => {
  const { q, domain, area, wichtigkeit, schlagwort, page } = req.query
  const result = listMemories({
    q: q || undefined,
    domain: domain || undefined,
    area: area || undefined,
    wichtigkeit: wichtigkeit || undefined,
    schlagwort: schlagwort || undefined,
    page: page ? Number(page) : 1,
  })
  res.json(result)
})

// GET /api/memories/:id — Einzelnes Memory
app.get('/api/memories/:id', (req, res) => {
  const memory = getMemory(req.params.id)
  if (!memory) return res.status(404).json({ error: 'Memory nicht gefunden' })
  res.json(memory)
})

// POST /api/memories — Anlegen
app.post('/api/memories', async (req, res) => {
  const { text, domain, area, wichtigkeit, schlagwoerter } = req.body || {}
  if (!text || !domain || !area) {
    return res.status(400).json({ error: 'text, domain und area sind Pflichtfelder' })
  }
  if (!['Privat', 'Beruf', 'Wissen'].includes(domain)) {
    return res.status(400).json({ error: 'domain muss Privat, Beruf oder Wissen sein' })
  }
  const embeddingBytes = await getEmbedding(text)
  const memory = insertMemory({ text, domain, area, wichtigkeit, schlagwoerter }, embeddingBytes)
  if (!embeddingBytes) {
    return res.status(201).json({ ...memory, _warning: 'Embedding fehlgeschlagen — semantische Suche veraltet' })
  }
  res.status(201).json(memory)
})

// PUT /api/memories/:id — Bearbeiten
app.put('/api/memories/:id', async (req, res) => {
  const existing = getMemory(req.params.id)
  if (!existing) return res.status(404).json({ error: 'Memory nicht gefunden' })

  const data = req.body || {}
  const textChanged = data.text !== undefined && data.text !== existing.text
  const embeddingBytes = textChanged ? await getEmbedding(data.text) : null

  const memory = updateMemory(req.params.id, data, embeddingBytes)
  if (textChanged && !embeddingBytes) {
    return res.json({ ...memory, _warning: 'Embedding fehlgeschlagen — semantische Suche veraltet' })
  }
  res.json(memory)
})

// DELETE /api/memories/:id — Löschen
app.delete('/api/memories/:id', (req, res) => {
  const existing = getMemory(req.params.id)
  if (!existing) return res.status(404).json({ error: 'Memory nicht gefunden' })
  deleteMemory(req.params.id)
  res.status(204).end()
})
```

```jsx
function MemoryDetail({ memory, isNew, onSave, onDelete, onCancel }) {
  const [form, setForm] = useState(memory || { text: '', domain: 'Privat', area: '', wichtigkeit: 'Wichtig', schlagwoerter: '' })
  const [saving, setSaving] = useState(false)
  const [deleting, setDeleting] = useState(false)

  useEffect(() => {
    setForm(memory || { text: '', domain: 'Privat', area: '', wichtigkeit: 'Wichtig', schlagwoerter: '' })
  }, [memory?.id, isNew])

  const handleSave = async () => {
    if (!form.text.trim() || !form.domain || !form.area.trim()) {
      toast('text, domain und area sind Pflichtfelder', 'error')
      return
    }
    setSaving(true)
    try {
      await onSave(form)
    } finally {
      setSaving(false)
    }
  }

  const handleDelete = async () => {
    if (!confirm('Memory endgültig löschen?')) return
    setDeleting(true)
    try {
      await onDelete(memory.id)
    } finally {
      setDeleting(false)
    }
  }

  if (!memory && !isNew) {
    return (
      <div className="flex-1 flex items-center justify-center" style={{ color: 'var(--hint)' }}>
        <div className="text-center">
          <Brain size={40} strokeWidth={1} className="mx-auto mb-3" style={{ color: 'var(--surface1)' }} />
          <p className="text-sm">Memory auswählen oder neu anlegen</p>
        </div>
      </div>
    )
  }

  return (
    <div className="flex-1 flex flex-col overflow-y-auto p-4 gap-3">
      <div className="flex justify-between items-center">
        <span className="text-sm font-semibold" style={{ color: 'var(--text)' }}>

              disabled={deleting}
              className="rounded px-3 py-1 text-xs font-medium"
              style={{ background: 'var(--red)', color: 'var(--base)', opacity: deleting ? 0.6 : 1 }}

              {deleting ? 'Löscht…' : 'Löschen'}

          value={form.text}
          onChange={e => setForm(f => ({ ...f, text: e.target.value }))}
          rows={5}
          className="rounded-lg px-3 py-2 text-sm outline-none resize-y w-full"
          style={{ background: 'var(--surface0)', color: 'var(--text)', fontSize: '14px', minHeight: '80px' }}
        />

          <label className="text-[10px] uppercase tracking-wide" style={{ color: 'var(--subtext0)' }}>Domain</label>
          <select
            value={form.domain}
            onChange={e => setForm(f => ({ ...f, domain: e.target.value }))}
            className="rounded-lg px-2 py-1.5 text-sm outline-none"

            <option value="">—</option>
            {WICHTIGKEITEN.map(w => <option key={w} value={w}>{w}</option>)}
          </select>
        </div>
      </div>

      <div className="flex flex-col gap-1">
        <label className="text-[10px] uppercase tracking-wide" style={{ color: 'var(--subtext0)' }}>Schlagwörter (kommagetrennt)</label>
        <input
          value={form.schlagwoerter || ''}
          onChange={e => setForm(f => ({ ...f, schlagwoerter: e.target.value }))}
          placeholder="Claude, Automatisierung, Heartbeat"
          className="rounded-lg px-2 py-1.5 text-sm outline-none"
          style={{ background: 'var(--surface0)', color: 'var(--text)', fontSize: '14px' }}
        />
      </div>

      {!isNew && memory && (
        <div className="text-[10px] pt-2 flex gap-4 flex-wrap" style={{ color: 'var(--hint)', borderTop: '1px solid var(--surface0)' }}>
          <span>ID: {memory.id.slice(0, 8)}</span>
          <span>Erstellt: {memory.created_at?.slice(0, 10)}</span>
          <span>Geändert: {memory.updated_at?.slice(0, 10)}</span>
          <span>Quelle: {memory.quelle_typ}</span>
        </div>
      )}
    </div>
  )
}

// ---- MemoryView (Haupt-Komponente) ----

export default function MemoryView() {
  const [filters, setFilters] = useState({ q: '', domain: '', area: '', wichtigkeit: '', schlagwort: '', page: 1 })
  const [result, setResult] = useState({ memories: [], total: 0, page: 1, pages: 1 })
  const [loading, setLoading] = useState(false)
  const [selectedId, setSelectedId] = useState(null)
  const [selectedMemory, setSelectedMemory] = useState(null)
  const [isNew, setIsNew] = useState(false)

  const fetchList = useCallback(async (f) => {
    setLoading(true)
    try {
      const params = new URLSearchParams()
      if (f.q) params.set('q', f.q)
      if (f.domain) params.set('domain', f.domain)
      if (f.area) params.set('area', f.area)
      if (f.wichtigkeit) params.set('wichtigkeit', f.wichtigkeit)
      if (f.schlagwort) params.set('schlagwort', f.schlagwort)
      params.set('page', f.page)
      const res = await fetch(`/api/memories?${params}`)
      if (!res.ok) throw new Error('Laden fehlgeschlagen')
      setResult(await res.json())
    } catch (e) {
      toast(e.message, 'error')
    } finally {
      setLoading(false)
    }
  }, [])

  useEffect(() => { fetchList(filters) }, [filters, fetchList])

  const handleSelect = async (id) => {
    setIsNew(false)
    setSelectedId(id)
    try {
      const res = await fetch(`/api/memories/${id}`)
      if (!res.ok) throw new Error('Laden fehlgeschlagen')
      setSelectedMemory(await res.json())
    } catch (e) {
      toast(e.message, 'error')
    }
  }

  const handleNew = () => {
    setSelectedId(null)
    setSelectedMemory(null)
    setIsNew(true)
  }

  const handleSave = async (form) => {
    try {
      const method = isNew ? 'POST' : 'PUT'
      const url = isNew ? '/api/memories' : `/api/memories/${selectedId}`
      const res = await fetch(url, {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(form),
      })
      if (!res.ok) {
        const err = await res.json()
        throw new Error(err.error || 'Speichern fehlgeschlagen')
      }
      const saved = await res.json()
      if (saved._warning) toast(saved._warning, 'info')
      else toast('Gespeichert', 'success')
      setIsNew(false)
      setSelectedId(saved.id)
      setSelectedMemory(saved)
      fetchList(filters)
    } catch (e) {
      toast(e.message, 'error')
    }
  }

  const handleDelete = async (id) => {
    try {
      const res = await fetch(`/api/memories/${id}`, { method: 'DELETE' })
      if (!res.ok) throw new Error('Löschen fehlgeschlagen')
      toast('Gelöscht', 'success')
      setSelectedId(null)
      setSelectedMemory(null)
      setIsNew(false)
      fetchList(filters)
    } catch (e) {
      toast(e.message, 'error')
    }
  }

  return (
    <div className="flex flex-col h-screen overflow-hidden">
      <FilterBar
        filters={filters}
        onChange={setFilters}
        onNew={handleNew}
      />
      <div className="flex flex-1 overflow-hidden" style={{ opacity: loading ? 0.7 : 1 }}>
        <MemoryList
          memories={result.memories}
          total={result.total}
          page={result.page}
          pages={result.pages}
          selectedId={selectedId}
          onSelect={handleSelect}
          onPageChange={p => setFilters(f => ({ ...f, page: p }))}
        />
        <MemoryDetail
          memory={selectedMemory}
          isNew={isNew}
          onSave={handleSave}
          onDelete={handleDelete}
          onCancel={() => setIsNew(false)}
        />
      </div>
    </div>
  )
}
```
